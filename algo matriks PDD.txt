cek MDP atau ready = jika MDP maka 2 hari setelahnya pin ready(SPK)(sabtu gak masuk)
		     jika Ready langsung tgl SPK
cek tanggal READY = pin merah(SPK)
cek Pembayaran = jika cash : pin kuning (disaat uang masuk/setelah SPK)
		 Jika Credit: ping kuning(dalam waktu 4 hari/sabtu gak masuk)
AFI(buka Faktur)=pin hitam

cek Tipe Mobil =	tanpa STNK : 	DR(delivery request)setelah AFI

					if(CKD & 6hari)tambah 6 hari dari DR untuk pin IN(mobil masuk plaza)
					else tambah 4 hari dari DR untuk pin IN(mobil masuk plaza)

					pin H-1(pemasangan aksesoris dll) setelah pin IN
					pin BSTB(delivery ke customer)setelah pin H-1

			dengan STNK:	if(CKD){
						tambah 10 hari dari AFI untuk pin H-1(sabtu gak masuk)
							
						if(6hari)pin DR kurang 4 hari -1 dari pin H-1(sabtu masuk)
						else pin DR kurang 4 hari -1 dari pin H-1(sabtu masuk)	
						
						}
					else {
						tambah 25 hari dari AFI untuk pin H-1(sabtu gak masuk)
						pin DR kurang 4 hari -1 dari pin H-1(sabtu masuk)
					     }
					pin IN kurang 1 hari dari pin H-1
					pin BSTB setelah pin H-1
